	const playersForCurrentGame = useMemo(() => {
		if (!Array.isArray(availablePlayers)) {
			return [];
		}
		if (
			!gameSessionState.selectedPlayerIds ||
			gameSessionState.selectedPlayerIds.length === 0
		) {
			return [];
		}
		const gamePlayers = availablePlayers.filter((player) =>
			gameSessionState.selectedPlayerIds.includes(player.id)
		);
		return gamePlayers;
	}, [availablePlayers, gameSessionState.selectedPlayerIds]);

	useEffect(() => {
		if (masterRoster) {
			setAvailablePlayers(masterRoster);
		}
		if (masterRosterError) {
			console.error(
				'[TanStack Query] Error loading master roster:',
				masterRosterError
			);
			setAvailablePlayers([]);
		}
	}, [masterRoster, masterRosterError, setAvailablePlayers]);

	useEffect(() => {
		if (isSupabaseUserLoading) {
			return;
		}
		if (supabaseUserError) {
			setSeasonsState([]);
			return;
		}
		if (!isSignedIn || !supabaseUserId) {
			setSeasonsState([]);
			return;
		}
		if (isLoadingSeasons) {
		} else if (seasonsError) {
			console.error(
				'[Seasons Effect] TanStack Query error loading seasons:',
				seasonsError
			);
			setSeasonsState([]);
		} else if (seasons) {
			setSeasonsState(Array.isArray(seasons) ? seasons : []);
		} else {
			setSeasonsState([]);
		}
	}, [
		isSignedIn,
		supabaseUserId,
		isSupabaseUserLoading,
		supabaseUserError,
		seasons,
		isLoadingSeasons,
		seasonsError,
	]);

	useEffect(() => {
		if (isLoadingTournaments) {
		}
		if (tournaments) {
			setTournamentsState(Array.isArray(tournaments) ? tournaments : []);
		}
		if (tournamentsError) {
			console.error(
				'[TanStack Query] Error loading tournaments:',
				tournamentsError
			);
			setTournamentsState([]);
		}
	}, [tournaments, isLoadingTournaments, tournamentsError]);

	useEffect(() => {
		if (availablePlayers && availablePlayers.length > 0) {
			setPlayersOnField((prevPlayersOnField) => {
				const nextPlayersOnField = prevPlayersOnField.map((fieldPlayer) => {
					const rosterPlayer = availablePlayers.find(
						(ap) => ap.id === fieldPlayer.id
					);
					if (rosterPlayer) {
						if (
							fieldPlayer.name !== rosterPlayer.name ||
							fieldPlayer.jerseyNumber !== rosterPlayer.jerseyNumber ||
							fieldPlayer.isGoalie !== rosterPlayer.isGoalie ||
							fieldPlayer.nickname !== rosterPlayer.nickname ||
							fieldPlayer.notes !== rosterPlayer.notes
						) {
							return {
								...fieldPlayer,
								name: rosterPlayer.name,
								jerseyNumber: rosterPlayer.jerseyNumber,
								isGoalie: rosterPlayer.isGoalie,
								nickname: rosterPlayer.nickname,
								notes: rosterPlayer.notes,
								receivedFairPlayCard:
									rosterPlayer.receivedFairPlayCard !== undefined
										? rosterPlayer.receivedFairPlayCard
										: fieldPlayer.receivedFairPlayCard,
							};
						}
					}
					return fieldPlayer;
				});

				if (
					JSON.stringify(prevPlayersOnField) !==
					JSON.stringify(nextPlayersOnField)
				) {
					saveStateToHistory({ playersOnField: nextPlayersOnField });
				}
				return nextPlayersOnField;
			});
		}
	}, [availablePlayers, saveStateToHistory, setPlayersOnField]);

	useEffect(() => {
		let intervalId: NodeJS.Timeout | null = null;
		const periodEndTimeSeconds =
			gameSessionState.currentPeriod * gameSessionState.periodDurationMinutes * 60;

		if (
			gameSessionState.isTimerRunning &&
			gameSessionState.gameStatus === 'inProgress'
		) {
			intervalId = setInterval(() => {
				const currentTime = gameSessionState.timeElapsedInSeconds;
				const potentialNewTime = currentTime + 1;

				if (potentialNewTime >= periodEndTimeSeconds) {
					clearInterval(intervalId!);
					if (
						gameSessionState.currentPeriod === gameSessionState.numberOfPeriods
					) {
						dispatchGameSession({
							type: 'END_PERIOD_OR_GAME',
							payload: {
								newStatus: 'gameEnd',
								finalTime: periodEndTimeSeconds,
							},
						});
					} else {
						dispatchGameSession({
							type: 'END_PERIOD_OR_GAME',
							payload: {
								newStatus: 'periodEnd',
								finalTime: periodEndTimeSeconds,
							},
						});
					}
				} else {
					dispatchGameSession({
						type: 'SET_TIMER_ELAPSED',
						payload: potentialNewTime,
					});
				}
			}, 1000);
		} else {
			if (intervalId) {
				clearInterval(intervalId);
			}
		}

		return () => {
			if (intervalId) {
				clearInterval(intervalId);
			}
		};
	}, [
		gameSessionState.isTimerRunning,
		gameSessionState.gameStatus,
		gameSessionState.currentPeriod,
		gameSessionState.periodDurationMinutes,
		gameSessionState.numberOfPeriods,
		gameSessionState.timeElapsedInSeconds,

	useEffect(() => {
		const loadInitialAppData = async () => {
			try {
				const oldRosterJson = await getLocalStorageItemAsync('availablePlayers');
				if (oldRosterJson) {
					await setLocalStorageItemAsync(MASTER_ROSTER_KEY, oldRosterJson);
					await removeLocalStorageItemAsync('availablePlayers');
				}
				const oldSeasonsJson = await getLocalStorageItemAsync('soccerSeasonsList');
				if (oldSeasonsJson) {
					await setLocalStorageItemAsync(SEASONS_LIST_KEY, oldSeasonsJson);
				}
			} catch (migrationError) {
				console.error('[EFFECT init] Error during data migration:', migrationError);
			}

			if (isLoadingSavedGames) {
				setIsLoadingGamesList(true);
			}
			if (allSavedGames) {
				setSavedGames(allSavedGames || {});
				setIsLoadingGamesList(false);
			}
			if (savedGamesError) {
				setLoadGamesListError(t('loadGameModal.errors.listLoadFailed'));
				setSavedGames({});
				setIsLoadingGamesList(false);
			}

			if (isLoadingCurrentGameId || isLoadingSavedGames) {
			} else {
				const lastGameIdSetting = currentGameId;
				const currentSavedGames = allSavedGames || {};

				if (
					lastGameIdSetting &&
					lastGameIdSetting !== DEFAULT_GAME_ID &&
					currentSavedGames[lastGameIdSetting]
				) {
					setCurrentGameStateId(lastGameIdSetting);
					setHasSkippedInitialSetup(true);
				} else {
					setCurrentGameStateId(DEFAULT_GAME_ID);
				}
			}

			if (
				!isLoadingMasterRoster &&
				!isLoadingSeasons &&
				!isLoadingTournaments &&
				!isLoadingSavedGames &&
				!isLoadingCurrentGameId
			) {
				setIsLoaded(true);
				setInitialLoadComplete(true);
			}
		};

		loadInitialAppData();
	}, [
		masterRoster, isLoadingMasterRoster, masterRosterError,
		seasons, isLoadingSeasons, seasonsError,
		tournaments, isLoadingTournaments, tournamentsError,
		allSavedGames, isLoadingSavedGames, savedGamesError,
		currentGameId, isLoadingCurrentGameId, currentGameIdError,
		t,

	const timerStateOnHideRef = useRef<{
		isRunning: boolean;
		elapsedSeconds: number;
		timestamp: number;
	} | null>(null);

	useEffect(() => {
		const handleVisibilityChange = () => {
			if (document.hidden) {
				if (
					gameSessionState.isTimerRunning &&
					gameSessionState.gameStatus === 'inProgress'
				) {
					timerStateOnHideRef.current = {
						isRunning: true,
						elapsedSeconds: gameSessionState.timeElapsedInSeconds,
						timestamp: Date.now(),
					};
					dispatchGameSession({ type: 'SET_TIMER_RUNNING', payload: false });
				}
			} else {
				if (timerStateOnHideRef.current && timerStateOnHideRef.current.isRunning) {
					const elapsedOfflineMs =
						Date.now() - timerStateOnHideRef.current.timestamp;
					const correctedElapsedSeconds =
						timerStateOnHideRef.current.elapsedSeconds + elapsedOfflineMs / 1000;

					dispatchGameSession({
						type: 'SET_TIMER_ELAPSED',
						payload: correctedElapsedSeconds,
					});

					if (gameSessionState.gameStatus === 'inProgress') {
						dispatchGameSession({ type: 'SET_TIMER_RUNNING', payload: true });
					}
					timerStateOnHideRef.current = null;
				}
			}
		};

		document.addEventListener('visibilitychange', handleVisibilityChange);
		return () => {
			document.removeEventListener('visibilitychange', handleVisibilityChange);
		};

		gameSessionState.isTimerRunning,
		gameSessionState.timeElapsedInSeconds,
		gameSessionState.gameStatus,

			setOpponents(gameData?.opponents || (isInitialDefaultLoad ? initialState.opponents : []));
			setDrawings(gameData?.drawings || (isInitialDefaultLoad ? initialState.drawings : []));

			const newHistoryState: AppState = {

				numberOfPeriods: gameData?.numberOfPeriods ?? initialGameSessionData.numberOfPeriods,

				completedIntervalDurations: gameData?.completedIntervalDurations ?? initialGameSessionData.completedIntervalDurations,
				lastSubConfirmationTimeSeconds: gameData?.lastSubConfirmationTimeSeconds ?? initialGameSessionData.lastSubConfirmationTimeSeconds,
				showPlayerNames: gameData?.showPlayerNames === undefined ? initialGameSessionData.showPlayerNames : gameData.showPlayerNames,
				selectedPlayerIds: gameData?.selectedPlayerIds ?? initialGameSessionData.selectedPlayerIds,
				gameEvents: gameData?.gameEvents ?? initialGameSessionData.gameEvents,
				playersOnField: gameData?.playersOnField || initialState.playersOnField,
				opponents: gameData?.opponents || initialState.opponents,
				drawings: gameData?.drawings || initialState.drawings,
				availablePlayers: masterRoster || availablePlayers,

			setHistoryIndex(0);
		},
		[availablePlayers, initialGameSessionData, masterRoster]

	useEffect(() => {

			return;
		}
		let gameToLoad: AppState | null = null;
		if (
			currentGameStateId &&
			currentGameStateId !== DEFAULT_GAME_ID &&
			savedGames[currentGameStateId]
		) {
			gameToLoad = savedGames[currentGameStateId] as AppState;
		}
		loadGameStateFromData(gameToLoad);
	}, [currentGameStateId, savedGames, initialLoadComplete, loadGameStateFromData]);

	useEffect(() => {
		const autoSave = async () => {
			if (
				isLoaded &&
				currentGameStateId &&
				currentGameStateId !== DEFAULT_GAME_ID
			) {
				try {
					const currentSnapshot = {
						...history[historyIndex],
						availablePlayers: masterRoster || availablePlayers,
					};

					if (token) {
						await utilSaveGame(token, currentGameStateId, currentSnapshot as AppState);
						await utilSaveCurrentGameIdSetting(token, currentGameStateId);
					}
				} catch (error) {
					console.error('Failed to auto-save state:', error);
					alert('Error saving game.');
				}
			}
		};
		autoSave();
	}, [
		isLoaded,
		currentGameStateId,
		playersOnField,
		opponents,
		drawings,
		availablePlayers,
		masterRoster,
		gameSessionState,
		history,
		historyIndex,
		token,
	]);

	useEffect(() => {

		}
	}, [initialLoadComplete, hasSkippedInitialSetup, currentGameStateId, isSignedIn]);

		},
		[setPlayersOnField]
	);

	const handlePlayerMoveEnd = useCallback(() => {
		saveStateToHistory({ playersOnField });
	}, [playersOnField, saveStateToHistory]);

	const handlePlayerRemove = useCallback(
		(playerId: string) => {
			const updatedPlayersOnField = playersOnField.filter(
				(p) => p.id !== playerId
			);
			setPlayersOnField(updatedPlayersOnField);
			saveStateToHistory({ playersOnField: updatedPlayersOnField });
		},
		[playersOnField, saveStateToHistory, setPlayersOnField]
	);

	const handleResetField = useCallback(() => {
		setPlayersOnField([]);
		setOpponents([]);
		setDrawings([]);
		saveStateToHistory({ playersOnField: [], opponents: [], drawings: [] });
	}, [saveStateToHistory, setDrawings, setOpponents, setPlayersOnField]);

	const handlePlayerDragStartFromBar = useCallback((playerInfo: Player) => {
		setDraggingPlayerFromBarInfo(playerInfo);
	}, []);

			} else {
				setDraggingPlayerFromBarInfo(playerInfo);
			}

	const handleTogglePlayerNames = () => {

			dispatchGameSession({ type: 'SET_GAME_DATE', payload: nextState.gameDate });

						nextState.completedIntervalDurations ?? [],
					lastSubConfirmationTimeSeconds:
						nextState.lastSubConfirmationTimeSeconds ?? 0,
					showPlayerNames: nextState.showPlayerNames,
					gameEvents: nextState.gameEvents,
					selectedPlayerIds: nextState.selectedPlayerIds,

					subIntervalMinutes: gameSessionState.subIntervalMinutes,

				type: 'SET_TIMER_RUNNING',
				payload: !gameSessionState.isTimerRunning,

	};

	const handleResetTimer = () => {

	const handleAddGoalEvent = (scorerId: string, assisterId?: string) => {
		const scorer = (masterRoster || availablePlayers).find(
			(p) => p.id === scorerId
		);
		const assister = assisterId
			? (masterRoster || availablePlayers).find((p) => p.id === assisterId)
			: undefined;

		if (!scorer) {
			console.error('Scorer not found!');
			return;
		}

		const newEvent: GameEvent = {
			id: `goal-${Date.now()}-${Math.random().toString(36).substring(2, 7)}`,
			type: 'goal',
			time: gameSessionState.timeElapsedInSeconds,
			scorerId: scorer.id,
			assisterId: assister?.id,
		};

		dispatchGameSession({ type: 'ADD_GAME_EVENT', payload: newEvent });
		dispatchGameSession({
			type: 'ADJUST_SCORE_FOR_EVENT',
			payload: { eventType: 'goal', action: 'add' },
		});
	};

	const handleLogOpponentGoal = (time: number) => {
		const newEvent: GameEvent = {
			id: `oppGoal-${Date.now()}-${Math.random().toString(36).substring(2, 7)}`,
			type: 'opponentGoal',
			time: time,
			scorerId: 'opponent',
		};

		dispatchGameSession({ type: 'ADD_GAME_EVENT', payload: newEvent });
		dispatchGameSession({
			type: 'ADJUST_SCORE_FOR_EVENT',
			payload: { eventType: 'opponentGoal', action: 'add' },
		});
		setIsGoalLogModalOpen(false);
	};

			type: updatedEvent.type,
			time: updatedEvent.time,
			scorerId: updatedEvent.scorerId,
			assisterId: updatedEvent.assisterId,

	const handleDeleteGameEvent = (goalId: string) => {
		const eventToDelete = gameSessionState.gameEvents.find(
			(e) => e.id === goalId
		);
		if (!eventToDelete) {
			return;
		}

		dispatchGameSession({ type: 'DELETE_GAME_EVENT', payload: goalId });
		if (eventToDelete.type === 'goal' || eventToDelete.type === 'opponentGoal') {
			dispatchGameSession({
				type: 'ADJUST_SCORE_FOR_EVENT',
				payload: { eventType: eventToDelete.type, action: 'delete' },
			});
		}
	};
